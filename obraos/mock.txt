// Effective stock = catalog qty minus all quantities committed in phases
function effectiveStock(catId) {
  const cat = CATALOG.find(c => c.id === catId);
  if (!cat) return 0;
  const committed = phases.reduce((sum, ph) => {
    const m = (ph.materials||[]).find(m => m.matId === catId);
    return sum + (m ? m.qty : 0);
  }, 0);
  return cat.qty - committed;
}

function stockLevel(catId) {
  const free = effectiveStock(catId);
  if (free <= 0)   return 'none';
  if (free <= 50)  return 'low';
  if (free <= 200) return 'mid';
  return 'ok';
}

function stockBadge(catId) {
  const lvl = stockLevel(catId);
  const map = {
    ok:   `<span style="color:var(--green);font-size:10px">✓ OK</span>`,
    mid:  `<span style="color:var(--accent);font-size:10px">~ MED</span>`,
    low:  `<span style="color:var(--red);font-size:10px">⚠ BAJO</span>`,
    none: `<span style="color:var(--red);font-size:10px;font-weight:700">✕ AGOTADO</span>`,
  };
  return map[lvl] || '';
}

            style="width:42px;background:var(--bg3);border:1px solid var(--border2);border-radius:4px;
                   color:var(--accent);font-family:'JetBrains Mono',monospace;font-size:11px;
                   padding:2px 3px;text-align:center;outline:none">
        </td>
        <td style="text-align:right;font-family:'JetBrains Mono',monospace;font-size:11px;
                   color:var(--text2);padding:7px 0;white-space:nowrap">${fmtCost(cost)}</td>
        <td style="text-align:center;padding:7px 3px">${stockBadge(m.matId)}</td>
        <td style="padding:7px 0 7px 3px">
          <button onclick="removeMatFromPhase(${currentPhase},${mi})"
            title="Quitar material"
            style="background:rgba(239,68,68,0.1);border:none;color:var(--red);
                   border-radius:4px;padding:2px 5px;font-size:10px;cursor:pointer">✕</button>
        </td>
      </tr>`;
    });

    html += `</tbody><tfoot><tr>
      <td colspan="2" style="padding:8px 0;font-size:9px;color:var(--text3);
          font-family:'JetBrains Mono',monospace;letter-spacing:1px">TOTAL FASE</td>
      <td colspan="3" style="text-align:right;font-family:'JetBrains Mono',monospace;
          font-size:13px;font-weight:700;color:var(--accent);padding:8px 0">${fmtCost(total)}</td>
    </tr></tfoot></table>`;
  }

  html += `</div>`;
  panel.innerHTML = html;
}

// HELPERS

function fmtCost(n) {
  return 'Q' + Math.round(n).toLocaleString('es-GT');
}
function formatDate(d) {
  if (!d) return '—';
  const [y,mo,dd] = d.split('-');
  return `${parseInt(dd)} ${['Ene','Feb','Mar','Abr','May','Jun','Jul','Ago','Sep','Oct','Nov','Dic'][parseInt(mo)-1]} ${y}`;
}
function dateDiff(s, e) {
  const d = Math.round((new Date(e)-new Date(s))/86400000);
  if (d < 0) return '';
  if (d < 30) return `${d} días`;
  if (d < 56) return `${Math.round(d/7)} semanas`;
  return `${Math.round(d/30)} meses`;
}

function selectPhase(idx) {
  currentPhase = idx;
  showPhase(idx);
  renderPanel();
}
function showPhase(idx) {
  const ph = phases[idx];
  if (!ph) return;
  for (const [name, obj] of Object.entries(houseParts)) {
    obj.visible = (ph.parts||[]).includes(name);
  }
  const lbl = document.getElementById('phase-label');
  if (lbl) lbl.textContent = ph.name;
}
function updatePhaseDate(idx, field, val) {
  phases[idx][field] = val;
  persistPhases();
  renderPanel();
}
function updateMatPct(phIdx, matIdx, val) {
  phases[phIdx].materials[matIdx].pct = Math.max(0, Math.min(100, parseInt(val)||0));
  persistPhases();
}
function removeMatFromPhase(phIdx, matIdx) {
  // Return qty to stock (conceptually — catalog qty stays, committed decreases)
  phases[phIdx].materials.splice(matIdx, 1);
  persistPhases();
  renderPanel();
}
function deletePhase(idx) {
  if (!confirm(`¿Eliminar la fase "${phases[idx].name}"?\nLos materiales asignados serán devueltos al inventario disponible.`)) return;
  phases.splice(idx, 1);
  currentPhase = Math.min(currentPhase, phases.length - 1);
  if (currentPhase < 0) currentPhase = 0;
  showPhase(currentPhase);
  persistPhases();
  renderPanel();
}
function reorderPhase(idx, dir) {
  const swap = idx + dir;
  if (swap < 0 || swap >= phases.length) return;
  [phases[idx], phases[swap]] = [phases[swap], phases[idx]];
  if (currentPhase === idx) currentPhase = swap;
  else if (currentPhase === swap) currentPhase = idx;
  showPhase(currentPhase);
  persistPhases();
  renderPanel();
}

  };
  document.getElementById('mm-stock-status').innerHTML = statusMap[lvl]||'';
  infoEl.style.display = 'block';

function updateCostPreview() {
  const id  = parseInt(document.getElementById('mm-mat-select').value);
  const qty = parseInt(document.getElementById('mm-qty').value) || 0;
  const prev = document.getElementById('mm-cost-preview');
  const warn = document.getElementById('mm-overstock-warn');
  if (!id || qty <= 0) { prev.style.display='none'; warn.style.display='none'; return; }
  const cat  = CATALOG.find(c => c.id === id);
  if (!cat) return;
  const free = effectiveStock(id);
  document.getElementById('mm-total-cost').textContent = fmtCost(cat.unitCost * qty);
  prev.style.display = 'block';
  if (qty > free) {
    warn.innerHTML = `⚠ La cantidad ingresada (<strong>${qty}</strong>) supera el stock libre (<strong>${free} ${cat.unit}</strong>). Se registrará el déficit.`;
    warn.style.display = 'block';
  } else {
    warn.style.display = 'none';
  }
}

function saveMatToPhase() {
  const phIdx = parseInt(document.getElementById('mm-phase-idx').value);
  const matId = parseInt(document.getElementById('mm-mat-select').value);
  const qty   = parseInt(document.getElementById('mm-qty').value) || 0;
  const pct   = parseInt(document.getElementById('mm-pct').value) || 0;
  if (!matId) { alert('Selecciona un material.'); return; }
  if (qty <= 0) { alert('Ingresa una cantidad válida.'); return; }

  const free = effectiveStock(matId);
  if (qty > free) {
    const cat = CATALOG.find(c => c.id === matId);
    if (!confirm(`La cantidad (${qty} ${cat?.unit}) supera el stock libre (${free} ${cat?.unit}).\n¿Continuar y registrar el déficit?`)) return;
  }

  phases[phIdx].materials = phases[phIdx].materials || [];
  phases[phIdx].materials.push({ matId, qty, pct });
  closeMatModal();
  persistPhases();
  renderPanel();
  // Refresh materials table if visible
  if (document.getElementById('screen-materials').classList.contains('active')) renderMaterialsTable();
}

let scene, camera, renderer, animId;
let rendererStarted = false;
let isDragging = false, prevMouse = {x:0,y:0};
let camTheta = 0.6, camPhi = 1.05, camRadius = 20;
let canvas;

function initThree() {
  rendererStarted = true;
  canvas = document.getElementById('three-canvas');
  const wrap = canvas.parentElement;
  scene = new THREE.Scene();
  scene.background = new THREE.Color(0x0d1117);
  scene.fog = new THREE.Fog(0x0d1117, 30, 60);
  const w = wrap.clientWidth, h = wrap.clientHeight;
  camera = new THREE.PerspectiveCamera(45, w/h, 0.1, 100);
  updateCamera();
  renderer = new THREE.WebGLRenderer({ canvas, antialias: true });
  renderer.setSize(w, h);
  renderer.shadowMap.enabled = true;
  renderer.shadowMap.type = THREE.PCFSoftShadowMap;
  // Lights
  scene.add(new THREE.AmbientLight(0xffffff, 0.45));
  const sun = new THREE.DirectionalLight(0xfff5e0, 1.2);
  sun.position.set(10, 20, 10); sun.castShadow = true;
  sun.shadow.mapSize.set(1024,1024); scene.add(sun);
  const fill = new THREE.DirectionalLight(0xaabfcc, 0.4);
  fill.position.set(-8,5,-8); scene.add(fill);
  scene.add(new THREE.GridHelper(30, 30, 0x1e2530, 0x1e2530));
  buildHouse();
  showPhase(currentPhase);
  // Mouse
  canvas.addEventListener('mousedown', e => { isDragging=true; prevMouse={x:e.clientX,y:e.clientY}; });
  window.addEventListener('mouseup',   () => isDragging=false);
  window.addEventListener('mousemove', e => {
    if (!isDragging) return;
    camTheta -= (e.clientX-prevMouse.x)*0.01;
    camPhi = Math.max(0.15, Math.min(1.5, camPhi+(e.clientY-prevMouse.y)*0.01));
    prevMouse = {x:e.clientX,y:e.clientY};
    updateCamera();
  });
  canvas.addEventListener('wheel', e => {
    camRadius = Math.max(6, Math.min(35, camRadius+e.deltaY*0.02));
    updateCamera();
  });
  // Touch
  let lastT = null;
  canvas.addEventListener('touchstart', e => { lastT=e.touches[0]; });
  canvas.addEventListener('touchmove',  e => {
    if (!lastT) return;
    const t = e.touches[0];
    camTheta -= (t.clientX-lastT.clientX)*0.01;
    camPhi = Math.max(0.15,Math.min(1.5,camPhi+(t.clientY-lastT.clientY)*0.01));
    lastT=t; updateCamera(); e.preventDefault();
  }, {passive:false});
  animate();
}

function updateCamera() {
  if (!camera) return;
  camera.position.x = camRadius*Math.sin(camPhi)*Math.sin(camTheta);
  camera.position.y = camRadius*Math.cos(camPhi);
  camera.position.z = camRadius*Math.sin(camPhi)*Math.cos(camTheta);
  camera.lookAt(0,3,0);
}

function buildHouse() {
  const T=0.3, BASE=0.95, WH=5.0, HW=5.0, HD=4.0;
  const Zf=+HD,Zb=-HD,Xr=+HW,Xl=-HW;
  const Zfc=Zf-T/2, Zbc=Zb+T/2, Xrc=Xr-T/2, Xlc=Xl+T/2;
  const IW=HW*2-T*2, ID=HD*2-T*2;

  const make = (geo,color,x,y,z) => {
    const m = new THREE.Mesh(geo, new THREE.MeshLambertMaterial({color}));
    m.position.set(x,y,z); m.castShadow=true; m.receiveShadow=true;
    return m;
  };
  const box = (w,h,d,col,x,y,z) => make(new THREE.BoxGeometry(w,h,d),col,x,y,z);
  const cyl = (r,h,col,x,y,z)   => make(new THREE.CylinderGeometry(r,r,h,8),col,x,y,z);
  const grp = name => { const g=new THREE.Group(); houseParts[name]=g; scene.add(g); return g; };

  // 1. Foundation
  const fg=grp('foundation');
  fg.add(box(HW*2,0.5,HD*2,MAT_COLORS.foundation,0,0.25,0));
  fg.add(box(HW*2,0.4,T,0x6b5c4d,0,0.7,Zfc));
  fg.add(box(HW*2,0.4,T,0x6b5c4d,0,0.7,Zbc));
  fg.add(box(T,0.4,HD*2,0x6b5c4d,Xrc,0.7,0));
  fg.add(box(T,0.4,HD*2,0x6b5c4d,Xlc,0.7,0));

  // 2. Columns (inside walls)
  const cg=grp('columns'); const CS=0.26;
  for(const[cx,cz]of[[Xlc,Zfc],[Xrc,Zfc],[Xlc,Zbc],[Xrc,Zbc],[0,Zfc],[0,Zbc],[Xlc,0],[Xrc,0]])
    cg.add(box(CS,WH,CS,MAT_COLORS.columns,cx,BASE+WH/2,cz));

  // 3. Slab (fits inside outer walls)
  const sg=grp('slab'); const slY=BASE+WH+0.15;
  sg.add(box(HW*2,0.22,HD*2,MAT_COLORS.slab,0,slY,0));
  const bY=slY-0.12;
  for(const xi of[-3,0,3]) sg.add(box(0.2,0.35,ID,0xa0b0bb,xi,bY,0));
  for(const zi of[-2,0,2]) sg.add(box(IW,0.35,0.2,0xa0b0bb,0,bY,zi));

  // 4. Walls
  const wg=grp('walls');
  const addFW=(cx,cy0,w,h)=>wg.add(box(w,h,T,MAT_COLORS.walls,cx,BASE+cy0+h/2,Zfc));
  addFW(0,0,HW*2,1.4);                             // base
  addFW(-4.25,1.4,1.5,1.45);                       // left pier
  addFW(-1.375,1.4,1.25,1.45);                     // pier L-win/door
  addFW(1.375,1.4,1.25,1.45);                      // pier door/R-win
  addFW(4.25,1.4,1.5,1.45);                        // right pier
  addFW(-2.875,2.3,4.25,0.55);                     // door spandrel L
  addFW(2.875,2.3,4.25,0.55);                      // door spandrel R
  addFW(0,2.85,HW*2,2.15);                         // top

  const addBW=(cx,cy0,w,h)=>wg.add(box(w,h,T,MAT_COLORS.walls,cx,BASE+cy0+h/2,Zbc));
  addBW(0,0,HW*2,1.4); addBW(-2.875,1.4,4.25,1.45); addBW(2.875,1.4,4.25,1.45); addBW(0,2.85,HW*2,2.15);

  const addLW=(cz,cy0,d,h)=>wg.add(box(T,h,d,MAT_COLORS.walls,Xlc,BASE+cy0+h/2,cz));
  addLW(0,0,HD*2,1.4); addLW(-2.375,1.4,3.25,1.45); addLW(2.375,1.4,3.25,1.45); addLW(0,2.85,HD*2,2.15);

  const addRW=(cz,cy0,d,h)=>wg.add(box(T,h,d,MAT_COLORS.walls,Xrc,BASE+cy0+h/2,cz));
  addRW(0,0,HD*2,1.4); addRW(-2.375,1.4,3.25,1.45); addRW(2.375,1.4,3.25,1.45); addRW(0,2.85,HD*2,2.15);

  // 5. Pipes (inside building)
  const pg=grp('pipes'); const pi=T+0.12;
  for(const[px,pz]of[[-3.5,Zf-pi],[3.5,Zf-pi],[-3.5,Zb+pi],[3.5,Zb+pi]])
    pg.add(cyl(0.05,3.5,MAT_COLORS.pipes,px,BASE+1.75,pz));
  for(const pz of[-2,0,2]){ const p=cyl(0.04,IW,MAT_COLORS.pipes,0,BASE+0.18,pz); p.rotation.z=Math.PI/2; pg.add(p); }

  // 6. Facade bands
  const facG=grp('facade'); const bC=0xc8a870;
  for(let fy=BASE+0.6;fy<BASE+WH-0.1;fy+=1.1){
    facG.add(box(HW*2+0.02,0.06,0.01,bC,0,fy,Zf+0.005));
    facG.add(box(HW*2+0.02,0.06,0.01,bC,0,fy,Zb-0.005));
    facG.add(box(0.01,0.06,HD*2+0.02,bC,Xl-0.005,fy,0));
    facG.add(box(0.01,0.06,HD*2+0.02,bC,Xr+0.005,fy,0));
  }

  // 7. Windows
  const winG=grp('windows');
  const WW=1.35,WH2=1.38,winCY=BASE+1.4+WH2/2+0.035;
  const addWin=(cx,cy,cz,ry)=>{
    const gl=box(WW,WH2,0.05,MAT_COLORS.windows,cx,cy,cz); gl.rotation.y=ry; winG.add(gl);
    const fr=box(WW+0.14,WH2+0.14,0.05,0x6a5040,cx,cy,cz); fr.rotation.y=ry; winG.add(fr);
    const mh=box(WW,0.05,0.07,0x5a4030,cx,cy,cz); mh.rotation.y=ry; winG.add(mh);
    const mv=box(0.05,WH2,0.07,0x5a4030,cx,cy,cz); mv.rotation.y=ry; winG.add(mv);
  };
  addWin(-2.75,winCY,Zf+0.02,0); addWin(2.75,winCY,Zf+0.02,0);
  addWin(0,winCY,Zb-0.02,0);
  addWin(Xl-0.02,winCY,-2.375,Math.PI/2); addWin(Xl-0.02,winCY,2.375,Math.PI/2);
  addWin(Xr+0.02,winCY,-2.375,Math.PI/2); addWin(Xr+0.02,winCY,2.375,Math.PI/2);

  // 8. Roof
  const rg=grp('roof'); const rfY=BASE+WH+0.28;
  const cone=make(new THREE.ConeGeometry(7.6,2.9,4),MAT_COLORS.roof,0,rfY+1.45,0);
  cone.rotation.y=Math.PI/4; rg.add(cone);
  rg.add(box(HW*2+1.3,0.1,HD*2+1.3,0x6d4c41,0,rfY+0.04,0));

  // 9. Door
  const dg=grp('door'); const DW=1.35,DH=2.25,dY=BASE+DH/2+0.02;
  dg.add(box(DW,DH,0.06,MAT_COLORS.door,0,dY,Zf+0.04));
  dg.add(box(DW+0.14,DH+0.1,0.04,0x3e2723,0,dY-0.04,Zf+0.03));
  dg.add(make(new THREE.SphereGeometry(0.07,8,6),0xd4a017,0.48,dY,Zf+0.08));

  // 10. Details
  const detG=grp('details'); const rfY2=BASE+WH+0.28;
  for(let i=0;i<3;i++) detG.add(box(1.9-i*0.15,0.13,0.42,0xa0a0a0,0,BASE-i*0.13-0.065,Zf+0.38+i*0.42));

// MARGIN PREVIEW in new project modal

function updateMarginPreview() {
  const budget  = parseFloat(document.getElementById('np-budget')?.value) || 0;
  const revenue = parseFloat(document.getElementById('np-revenue')?.value) || 0;
  const ovhd    = (parseFloat(document.getElementById('np-ovhd')?.value) || 0) / 100;
  const cont    = (parseFloat(document.getElementById('np-cont')?.value) || 0) / 100;
  const prev = document.getElementById('np-margin-preview');
  if (!prev) return;
  if (!budget || !revenue) { prev.style.display='none'; return; }
  const totalCost = budget * (1 + ovhd + cont);
  const margin = (revenue - totalCost) / revenue;
  const utilidad = revenue - totalCost;
  const mVal = document.getElementById('np-margin-val');
  mVal.textContent = `${(margin*100).toFixed(1)}%  (Q${Math.round(utilidad).toLocaleString('es-GT')})`;
  mVal.style.color = margin > 0 ? 'var(--green)' : 'var(--red)';
  prev.style.display = 'block';
}
// Wire up live preview
setTimeout(() => {
  ['np-budget','np-revenue','np-ovhd','np-cont'].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.addEventListener('input', updateMarginPreview);
  });
}, 500);

// EXCEL EXPORT  — uses SheetJS (xlsx) loaded from CDN
// Generates a real .xlsx with 4 sheets from current app state
// ═══════════════════════════════════════════════════════════════════
function exportExcel() {
  // Lazy-load SheetJS
  if (typeof XLSX === 'undefined') {
    const s = document.createElement('script');
    s.src = 'https://cdnjs.cloudflare.com/ajax/libs/xlsx/0.18.5/xlsx.full.min.js';
    s.onload = () => _doExport();
    document.head.appendChild(s);
  } else {
    _doExport();
  }
}

  const contPct = (parseFloat(document.getElementById('np-cont')?.value)   || 5)  / 100;
  const marginObj = sale > 0 ? (sale - budget*(1+ovhdPct+contPct)) / sale : 0.22;

  const phCost  = ph => (ph.materials||[]).reduce((s,m)=>{const c=CATALOG.find(x=>x.id===m.matId);return s+(c?c.unitCost*m.qty:0);},0);
  const phExec  = ph => (ph.materials||[]).reduce((s,m)=>{const c=CATALOG.find(x=>x.id===m.matId);return s+(c?c.unitCost*m.qty*(m.pct/100):0);},0);
